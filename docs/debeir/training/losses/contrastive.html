 16class SupConLoss(nn.Module):
 17    """Supervised Contrastive Learning: https://arxiv.org/pdf/2004.11362.pdf.
 18    It also supports the unsupervised contrastive loss in SimCLR"""
 19
 20    # def __init__(self, temperature=0.07, contrast_mode='all',
 21    #             base_temperature=0.07):
 22    def __init__(self, temperature=1.0, contrast_mode='all',
 23                 base_temperature=1.0):
 24        super(SupConLoss, self).__init__()
 25        self.temperature = temperature
 26        self.base_temperature = base_temperature
 27        self.contrast_mode = contrast_mode
 28
 29    def forward(self, features, labels=None, mask=None):
 30        """Compute loss for model. If both `labels` and `mask` are None,
 31        it degenerates to SimCLR unsupervised loss:
 32        https://arxiv.org/pdf/2002.05709.pdf
 33        Args:
 34            features: hidden vector of shape [bsz, n_views, ...].
 35            labels: ground truth of shape [bsz].
 36            mask: contrastive mask of shape [bsz, bsz], mask_{i,j}=1 if sample j
 37                has the same class as sample i. Can be asymmetric.
 38        Returns:
 39            A loss scalar.
 40        """
 41        device = (torch.device('cuda')
 42                  if features.is_cuda
 43                  else torch.device('cpu'))
 44
 45        if len(features.shape) < 3:
 46            raise ValueError('`features` needs to be [bsz, n_views, ...],'
 47                             'at least 3 dimensions are required')
 48        if len(features.shape) > 3:
 49            features = features.view(features.shape[0], features.shape[1], -1)
 50
 51        batch_size = features.shape[0]
 52        if labels is not None and mask is not None:
 53            raise ValueError('Cannot define both `labels` and `mask`')
 54        elif labels is None and mask is None:
 55            mask = torch.eye(batch_size, dtype=torch.float32).to(device)
 56        elif labels is not None:
 57            labels = labels.contiguous().view(-1, 1)
 58            if labels.shape[0] != batch_size:
 59                raise ValueError('Num of labels does not match num of features')
 60            mask = torch.eq(labels, labels.T).float().to(device)
 61        else:
 62            mask = mask.float().to(device)
 63
 64        contrast_count = features.shape[1]
 65        contrast_feature = torch.cat(torch.unbind(features, dim=1), dim=0)
 66        if self.contrast_mode == 'one':
 67            anchor_feature = features[:, 0]
 68            anchor_count = 1
 69        elif self.contrast_mode == 'all':
 70            anchor_feature = contrast_feature
 71            anchor_count = contrast_count
 72        else:
 73            raise ValueError('Unknown mode: {}'.format(self.contrast_mode))
 74
 75        # compute logits
 76        anchor_dot_contrast = torch.div(
 77            torch.matmul(anchor_feature, contrast_feature.T),
 78            self.temperature)
 79        # for numerical stability
 80        logits_max, _ = torch.max(anchor_dot_contrast, dim=1, keepdim=True)
 81        logits = anchor_dot_contrast - logits_max.detach()
 82
 83        # tile mask
 84        mask = mask.repeat(anchor_count, contrast_count)
 85        # mask-out self-contrast cases
 86        logits_mask = torch.scatter(
 87            torch.ones_like(mask),
 88            1,
 89            torch.arange(batch_size * anchor_count).view(-1, 1).to(device),
 90            0
 91        )
 92        mask = mask * logits_mask
 93
 94        # compute log_prob
 95        exp_logits = torch.exp(logits) * logits_mask
 96        log_prob = logits - torch.log(exp_logits.sum(1, keepdim=True))
 97
 98        # compute mean of log-likelihood over positive
 99        mean_log_prob_pos = (mask * log_prob).sum(1) / mask.sum(1)
100
101        # loss
102        loss = - (self.temperature / self.base_temperature) * mean_log_prob_pos
103        loss = loss.view(anchor_count, batch_size).mean()
104
105        return loss

117class ContrastiveSentLoss(nn.Module):
118    """
119    Contrastive loss. Expects as input two texts and a label of either 0 or 1. If the label == 1, then the distance between the
120    two embeddings is reduced. If the label == 0, then the distance between the embeddings is increased.
121    Further information: http://yann.lecun.com/exdb/publis/pdf/hadsell-chopra-lecun-06.pdf
122    :param model: SentenceTransformer model
123    :param distance_metric: Function that returns a distance between two emeddings. The class SiameseDistanceMetric contains pre-defined metrices that can be used
124    :param margin: Negative samples (label == 0) should have a distance of at least the margin value.
125    :param size_average: Average by the size of the mini-batch.
126    Example::
127        from sentence_transformers import SentenceTransformer, LoggingHandler, losses, InputExample
128        from torch.utils.data import DataLoader
129        model = SentenceTransformer('all-MiniLM-L6-v2')
130        train_examples = [
131            InputExample(texts=['This is a positive pair', 'Where the distance will be minimized'], label=1),
132            InputExample(texts=['This is a negative pair', 'Their distance will be increased'], label=0)]
133        train_dataloader = DataLoader(train_examples, shuffle=True, batch_size=2)
134        train_loss = losses.ContrastiveLoss(model=model)
135        model.fit([(train_dataloader, train_loss)], show_progress_bar=True)
136    """
137
138    def __init__(self, model, distance_metric=SiameseDistanceMetric.COSINE_DISTANCE,
139                 margin: float = 0.5, size_average: bool = True):
140        super(ContrastiveSentLoss, self).__init__()
141        self.distance_metric = distance_metric
142        self.margin = margin
143        self.model = model
144        self.size_average = size_average
145
146    def get_config_dict(self):
147        distance_metric_name = self.distance_metric.__name__
148        for name, value in vars(SiameseDistanceMetric).items():
149            if value == self.distance_metric:
150                distance_metric_name = "SiameseDistanceMetric.{}".format(name)
151                break
152
153        return {'distance_metric': distance_metric_name, 'margin': self.margin, 'size_average': self.size_average}
154
155    def forward(self, sentence_features: Iterable[Dict[str, Tensor]], labels: Tensor):
156        reps = [self.model(sentence_feature)['sentence_embedding'] for sentence_feature in sentence_features]
157        assert len(reps) == 2
158        rep_anchor, rep_other = reps
159        distances = self.distance_metric(rep_anchor, rep_other)
160        losses = 0.5 * (labels.float() * distances.pow(2) + (1 - labels).float() * F.relu(self.margin - distances).pow(2))
161        return losses.mean() if self.size_average else losses.sum()
